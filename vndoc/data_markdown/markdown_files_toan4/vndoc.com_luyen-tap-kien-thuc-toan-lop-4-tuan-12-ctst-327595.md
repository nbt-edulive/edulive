18 × 7 = 126

Ta có : a + a + 2 + a + 4 + a + 6 + a + 8 + a + 10 + a + 12 = 126
Hay 7 × a + 42 = 126
7 × a = 126 - 42 =
7 × a = 84
a = 84 : 7
a = 12
